# Basis Functions for Regression
curve_fit <- function(X) {
  if (X < 1) {
    return(1 + X) 
  } else {
    return(1 + X - 2 * (X - 1)^2) 
  }
}


X <- seq(-2, 2, by = 0.01)
Y <- sapply(X, curve_fit)


plot(X, Y, type = "l", col = "blue", lwd = 2, xlab = "X", ylab = "Y",
     main = "Linear Regression with Basis Functions")
abline(v = 1, col = "red", lty = 2)  
points(c(-2, 0, 1, 2), sapply(c(-2, 0, 1, 2), curve_fit), col = "black", pch = 16)


text(-2, curve_fit(-2), labels = round(curve_fit(-2), 2), pos = 3)
text(0, curve_fit(0), labels = round(curve_fit(0), 2), pos = 3)
text(1, curve_fit(1), labels = round(curve_fit(1), 2), pos = 3)
text(2, curve_fit(2), labels = round(curve_fit(2), 2), pos = 3)
